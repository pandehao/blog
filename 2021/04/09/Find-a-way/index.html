Find a way

2021年04月09日 737 字 大概 4 分钟

HDU - 2612

Pass a year learning in Hangzhou, yifenfei arrival hometown Ningbo at finally. Leave Ningbo one year, yifenfei have many people to meet. Especially a good friend Merceki. Yifenfei’s home is at the countryside, but Merceki’s home is in the center of city. So yifenfei made arrangements with Merceki to meet at a KFC. There are many KFC in Ningbo, they want to choose one that let the total time to it be most smallest. Now give you a Ningbo map, Both yifenfei and Merceki can move up, down ,left, right to the adjacent road by cost 11 minutes.

Input

The input contains multiple test cases.
Each test case include, first two integers n, m. (2<=n,m<=200).
Next n lines, each line included m character.
‘ Y ’ express yifenfei initial position.
‘M’ express Merceki initial position.
‘#’ forbid road;
‘.’ Road.
‘@’ KCF

Output

For each test case output the minimum total time that both yifenfei and Merceki to arrival one of KFC.You may sure there is always have a KFC that can let them meet.

Sample Input
4 4
Y.#@
....
.#..
@..M
4 4
Y.#@
....
.#..
@#.M
5 5
Y..@.
.#...
.#...
@..M.
#...#
Sample Output
66
88
66
Accepet Code
#include <bits/stdc++.h>
#define ll long long
using namespace std;
char mz[210][210];
int vis[210][210];
int n,m;
int dx[4]={0,0,1,-1};
int dy[4]={1,-1,0,0};

int ta[210][210];
int tb[210][210];
struct node{
	int x,y;
	int step;
};

struct P{
	int x,y;
};
P Y,M;

bool check(int x,int y){
	if(x>=0&&x<n&&y>=0&&y<m&&mz[x][y]!='#')
		return true;
	return false;
}

void bfsa(int x,int y){
	memset(ta,0,sizeof(ta));
	memset(vis,0,sizeof(vis));
	queue<node>q;
	node str;
	str.x=x;
	str.y=y;
	str.step=0;
	q.push(str);
	while(!q.empty()){
		node next;
		str=q.front();
		q.pop();
		if(mz[str.x][str.y]=='@'){
			ta[str.x][str.y]=str.step;
		}
		for(int i=0;i<4;i++){
			next.x= str.x+dx[i];
			next.y= str.y+dy[i];
			next.step=str.step+1;
			if(check(next.x,next.y)&&vis[next.x][next.y]==0){
				vis[next.x][next.y]=1;
				q.push(next);
			}
		}
	}
}

void bfsb(int x,int y){
	memset(tb,0,sizeof(tb));
	memset(vis,0,sizeof(vis));
	queue<node>q;
	node str;
	str.x=x;
	str.y=y;
	str.step=0;
	q.push(str);
	while(!q.empty()){
		node next;
		str=q.front();
		q.pop();
		if(mz[str.x][str.y]=='@'){
			tb[str.x][str.y]=str.step;
		}
		for(int i=0;i<4;i++){
			next.x= str.x+dx[i];
			next.y= str.y+dy[i];
			next.step=str.step+1;
			if(check(next.x,next.y)&&vis[next.x][next.y]==0){
				vis[next.x][next.y]=1;
				q.push(next);
			}
		}
	}
}

int main(){
	while(cin>>n>>m){
		memset(mz,0,sizeof(mz));
		for(int i=0;i<n;i++){
			for(int j=0;j<m;j++){
				cin>>mz[i][j];
				if(mz[i][j]=='Y'){
					Y.x=i;
					Y.y=j;
				}
				if(mz[i][j]=='M'){
					M.x=i;
					M.y=j;
				}
			}
		}
		bfsa(Y.x,Y.y);
		bfsb(M.x,M.y);
		int ans=1e9;
		for(int i=0;i<n;i++){
			for(int j=0;j<m;j++){
				if(mz[i][j]=='@'&&ta[i][j]!=0&&tb[i][j]!=0){
					ans=min(ans,11*(ta[i][j]+tb[i][j]));
				}
			}
		}
		cout<<ans<<endl;
	}	
}
